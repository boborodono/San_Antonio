CREATE TABLE stroke_data_cleaned (
     gender VARCHAR(6) NOT NULL,
     age INT NOT NULL,
	 hypertension INT NOT NULL,
	 heart_disease INT NOT NULL,
	 ever_married VARCHAR(3) NOT NULL,
	 work_type VARCHAR(13) NOT NULL,
	 residence_type VARCHAR(5) NOT NULL,
	 avg_glucose_level DECIMAL(5,2) NOT NULL,
	 bmi DECIMAL(3,1) NOT NULL,
	 smoking_status VARCHAR(14),
	 stroke INT NOT NULL
);
